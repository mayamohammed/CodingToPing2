function activateService(key, tabEl) {
  // Update hero cards
  document.querySelectorAll('.service-card').forEach(c => {
    c.classList.toggle('active', c.dataset.service === key);
  });
  // Update tabs
  if (tabEl) {
    document.querySelectorAll('.service-tab').forEach(t => t.classList.remove('active'));
    tabEl.classList.add('active');
  } else {
    document.querySelectorAll('.service-tab').forEach((t, i) => {
      const keys = ['sites','apps','ai','chat','dash'];
      t.classList.toggle('active', keys[i] === key);
    });
  }
  // Render features
  const grid = document.getElementById('feature-grid');
  const features = serviceData[key].features;
  grid.innerHTML = features.map(f => `
    <div class="feature-card">
      <div class="feature-icon">${f.icon}</div>
      <div class="feature-title">${f.title}</div>
      <div class="feature-desc">${f.desc}</div>
    </div>
  `).join('');
  grid.style.opacity = '0';
  grid.style.transform = 'translateY(16px)';
  setTimeout(() => {
    grid.style.transition = 'opacity .35s, transform .35s';
    grid.style.opacity = '1';
    grid.style.transform = 'none';
  }, 20);
  // Scroll to services section smoothly
  document.getElementById('services').scrollIntoView({ behavior: 'smooth', block: 'start' });
}
